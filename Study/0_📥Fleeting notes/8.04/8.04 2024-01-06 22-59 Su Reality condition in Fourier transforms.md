$$
\begin{aligned}

k \rightarrow-k & =\frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} \Phi^{*}(-k) e^{i k x} d k \\
& \stackrel{?}{=} \frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} \Phi(k) e^{i k x} d k=\Psi(x, 0)

$$
